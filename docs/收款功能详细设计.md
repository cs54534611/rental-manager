# 收款功能对接 - 详细设计

## 功能概述

为租房管理系统添加收款功能，支持租金在线支付、账单生成、支付记录管理。

## 一、数据库设计

### 1. 新增表：支付渠道 (`payment_channels`)

```sql
CREATE TABLE IF NOT EXISTS payment_channels (
  id INT PRIMARY KEY AUTO_INCREMENT,
  channel_type VARCHAR(20) NOT NULL COMMENT '支付渠道: wechat/alipay/bank',
  channel_name VARCHAR(50) NOT NULL COMMENT '渠道名称',
  app_id VARCHAR(100) COMMENT '应用ID',
  mch_id VARCHAR(50) COMMENT '商户号',
  api_key VARCHAR(200) COMMENT 'API密钥',
  cert_path VARCHAR(200) COMMENT '证书路径',
  is_enabled TINYINT DEFAULT 1 COMMENT '是否启用',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
```

### 2. 新增表：支付记录 (`payments`)

```sql
CREATE TABLE IF NOT EXISTS payments (
  id INT PRIMARY KEY AUTO_INCREMENT,
  payment_no VARCHAR(30) UNIQUE NOT NULL COMMENT '支付流水号',
  rental_id INT NOT NULL COMMENT '关联租金ID',
  tenant_id INT NOT NULL COMMENT '租客ID',
  house_id INT NOT NULL COMMENT '房源ID',
  amount DECIMAL(10,2) NOT NULL COMMENT '支付金额',
  channel VARCHAR(20) NOT NULL COMMENT '支付渠道',
  transaction_id VARCHAR(100) COMMENT '第三方交易号',
  status VARCHAR(20) DEFAULT 'pending' COMMENT 'pending/success/failed/refunded',
  pay_time DATETIME COMMENT '支付时间',
  notify_data JSON COMMENT '回调数据',
  remark TEXT COMMENT '备注',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_rental_id (rental_id),
  INDEX idx_status (status),
  INDEX idx_tenant_id (tenant_id)
);
```

### 1. 后端 - 支付控制器 (`server/src/routes/payments.js`)

```javascript
const express = require('express');
const router = express.Router();
const db = require('../db');
const { v4: uuidv4 } = require('uuid');

// 创建支付订单
router.post('/create', async (req, res) => {
  const { rental_id, channel, tenant_id } = req.body;
  
  // 查询租金信息
  const [rental] = await db.query(
    'SELECT * FROM rentals WHERE id = ? AND is_deleted = 0',
    [rental_id]
  );
  
  if (!rental) {
    return res.json({ code: 1, message: '租金不存在' });
  }
  
  // 生成支付流水号
  const payment_no = `PAY${Date.now()}${Math.random().toString(36).slice(2, 8)}`;
  
  // 创建支付记录
  await db.query(
    `INSERT INTO payments (payment_no, rental_id, tenant_id, house_id, amount, channel, status)
     VALUES (?, ?, ?, ?, ?, ?, 'pending')`,
    [payment_no, rental_id, tenant_id, rental.house_id, rental.receivable, channel]
  );
  
  // 根据渠道调用对应支付接口
  let payParams;
  if (channel === 'wechat') {
    payParams = await createWechatPay({ payment_no, amount: rental.receivable });
  } else if (channel === 'alipay') {
    payParams = await createAlipay({ payment_no, amount: rental.receivable });
  }
  
  res.json({ code: 0, data: payParams });
});

// 支付回调
router.post('/notify', async (req, res) => {
  const { channel } = req.body;
  
  let result;
  if (channel === 'wechat') {
    result = await verifyWechatNotify(req.body);
  } else if (channel === 'alipay') {
    result = await verifyAlipayNotify(req.body);
  }
  
  if (result.success) {
    // 更新支付状态
    await db.query(
      `UPDATE payments SET status = 'success', pay_time = NOW(), 
       transaction_id = ? WHERE payment_no = ?`,
      [result.transaction_id, result.payment_no]
    );
    
    // 更新租金状态
    await db.query(
      `UPDATE rentals SET status = 1, actual = receivable, 
       paid_date = CURDATE() WHERE id = ?`,
      [result.rental_id]
    );
  }
  
  res.send('success');
});

// 退款
router.post('/refund', async (req, res) => {
  const { payment_id, reason } = req.body;
  
  const [payment] = await db.query('SELECT * FROM payments WHERE id = ?', [payment_id]);
  
  // 调用退款接口
  let refundResult;
  if (payment.channel === 'wechat') {
    refundResult = await wechatRefund(payment);
  } else if (payment.channel === 'alipay') {
    refundResult = await alipayRefund(payment);
  }
  
  if (refundResult.success) {
    await db.query(
      "UPDATE payments SET status = 'refunded', remark = ? WHERE id = ?",
      [reason, payment_id]
    );
  }
  
  res.json({ code: 0, data: refundResult });
});

module.exports = router;
```

### 2. 后端 - 微信支付封装 (`server/src/payment/wechat.js`)

```javascript
const axios = require('axios');
const crypto = require('crypto');

class WechatPay {
  constructor(config) {
    this.appid = config.appid;
    this.mch_id = config.mch_id;
    this.api_key = config.api_key;
  }
  
  // 生成签名
  sign(params) {
    const stringA = Object.keys(params)
      .filter(key => params[key])
      .sort()
      .map(key => `${key}=${params[key]}`)
      .join('&');
    const stringSignTemp = `${stringA}&key=${this.api_key}`;
    return crypto.createHash('md5')
      .update(stringSignTemp, 'utf8')
      .digest('hex')
      .toUpperCase();
  }
  
  // 创建支付订单
  async createOrder({ payment_no, amount, description, openid }) {
    const params = {
      appid: this.appid,
      mch_id: this.mch_id,
      nonce_str: crypto.randomUUID(),
      body: description,
      out_trade_no: payment_no,
      total_fee: Math.round(amount * 100), // 转为分
      spbill_create_ip: '127.0.0.1',
      trade_type: 'JSAPI',
      openid
    };
    params.sign = this.sign(params);
    
    const response = await axios.post(
      'https://api.mch.weixin.qq.com/pay/unifiedorder',
      this.toXML(params)
    );
    
    return this.parseXML(response.data);
  }
  
  // 小程序调起支付
  async getPayParams(prepay_id) {
    const params = {
      appId: this.appid,
      timeStamp: Math.floor(Date.now() / 1000).toString(),
      nonceStr: crypto.randomUUID(),
      package: `prepay_id=${prepay_id}`,
      signType: 'MD5'
    };
    params.paySign = this.sign(params);
    return params;
  }
  
  toXML(obj) {
    return '<xml>' + Object.keys(obj)
      .map(k => `<${k}>${obj[k]}</${k}>`)
      .join('') + '</xml>';
  }

  }
}

module.exports = WechatPay;
```

### 3. 前端 - 支付页面 (`miniprogram/pages/rentals/pay.js`)

```javascript
Page({
  data: {
    rental: null,
    channels: [
      { id: 'wechat', name: '微信支付', icon: '/images/wechat.png' },
      { id: 'alipay', name: '支付宝', icon: '/images/alipay.png' }
    ],
    selectedChannel: 'wechat'
  },
  
  onLoad(options) {
    const { rental_id } = options;
    this.getRentalDetail(rental_id);
  },
  
  async getRentalDetail(id) {
    const res = await wx.request({
      url: `${getApp().globalData.baseUrl}/api/rentals/${id}`
    });
    this.setData({ rental: res.data.data });
  },
  
  selectChannel(e) {
    this.setData({ 
      selectedChannel: e.currentTarget.dataset.id 
    });
  },
  
  async handlePay() {
    const { rental, selectedChannel } = this.data;
    
    // 创建支付订单
    const createRes = await wx.request({
      url: `${getApp().globalData.baseUrl}/api/payments/create`,
      method: 'POST',
      data: {
        rental_id: rental.id,
        channel: selectedChannel,
        tenant_id: getApp().globalData.userId
      }
    });
    
    if (createRes.data.code !== 0) {
      wx.showToast({ title: createRes.data.message });
      return;
    }
    
    const payParams = createRes.data.data;
    
    // 调起支付
    if (selectedChannel === 'wechat') {
      wx.requestPayment({
        ...payParams,
        success: () => {
          wx.showToast({ title: '支付成功' });
          this.updateRentalStatus(rental.id);
        },
        fail: (err) => {
          wx.showToast({ title: '支付失败' });
        }
      });
    } else if (selectedChannel === 'alipay') {
      // 支付宝小程序支付
      my.tradePay({
        orderStr: payParams.orderStr,
        success: () => {
          wx.showToast({ title: '支付成功' });
        },
        fail: (err) => {
          wx.showToast({ title: '支付失败' });
        }
      });
    }
  },
  
  async updateRentalStatus(id) {
    await wx.request({
      url: `${getApp().globalData.baseUrl}/api/rentals/${id}`,
      method: 'PUT',
      data: { status: 1 }
    });
  }
});
```

## 六、安全注意事项

1. **敏感信息加密存储**
   - 商户API密钥需加密存储
   - 证书文件需妥善保管

2. **回调验签**
   - 微信/支付宝回调必须验签
   - 防止伪造回调

3. **幂等性处理**
   - 同一订单多次支付需处理
   - 回调重复推送需去重

4. **金额校验**
   - 支付金额必须与订单金额一致
   - 后端需再次校验
